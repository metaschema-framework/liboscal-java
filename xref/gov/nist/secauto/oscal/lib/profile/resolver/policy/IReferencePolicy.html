6   package gov.nist.secauto.oscal.lib.profile.resolver.policy;
7   
8   import gov.nist.secauto.metaschema.core.metapath.item.node.IModelNodeItem;
9   import gov.nist.secauto.oscal.lib.profile.resolver.ProfileResolutionEvaluationException;
10  
11  import edu.umd.cs.findbugs.annotations.NonNull;
12  
13  public interface IReferencePolicy<T> {
14    @NonNull
15    IReferencePolicy<Object> IGNORE_POLICY = new IReferencePolicy<>() {
16  
17      @Override
18      public boolean handleReference(
19          @NonNull IModelNodeItem<?, ?> contextItem,
20          @NonNull Object reference,
21          @NonNull ReferenceCountingVisitor.Context referenceVisitorContext) {
22        return true;
23      }
24    };
25  
26    /**
27     * Get a reference policy that will ignore processing the reference.
28     *
29     * @param <T>
30     *          the type of the reference object
31     * @return the policy
32     */
33    @SuppressWarnings("unchecked")
34    @NonNull
35    static <T> IReferencePolicy<T> ignore() {
36      return (IReferencePolicy<T>) IGNORE_POLICY;
37    }
38  
39    /**
40     * Handle the provided {@code reference}.
41     *
42     * @param contextItem
43     *          the nodes containing the reference
44     * @param reference
45     *          the reference object to process
46     * @param referenceVisitorContext
47     *          used to lookup and resolve items
48     * @return {@code true} if the reference was handled, or {@code false} otherwise
49     * @throws ProfileResolutionEvaluationException
50     *           if there was an error handing the reference
51     */
52    boolean handleReference(
53        @NonNull IModelNodeItem<?, ?> contextItem,
54        @NonNull T reference,
55        @NonNull ReferenceCountingVisitor.Context referenceVisitorContext);
56  }
