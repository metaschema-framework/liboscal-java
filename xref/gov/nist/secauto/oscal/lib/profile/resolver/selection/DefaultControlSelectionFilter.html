8   import gov.nist.secauto.metaschema.core.util.ObjectUtils;
9   import gov.nist.secauto.oscal.lib.model.Matching;
10  import gov.nist.secauto.oscal.lib.model.control.catalog.IControl;
11  import gov.nist.secauto.oscal.lib.model.control.profile.IProfileSelectControlById;
12  import gov.nist.secauto.oscal.lib.profile.resolver.ProfileResolutionEvaluationException;
13  
14  import org.apache.commons.lang3.tuple.Pair;
15  import org.apache.logging.log4j.LogManager;
16  import org.apache.logging.log4j.Logger;
17  
18  import java.util.Collections;
19  import java.util.List;
20  import java.util.Objects;
21  import java.util.Set;
22  import java.util.regex.Pattern;
23  import java.util.stream.Collectors;
24  
25  import edu.umd.cs.findbugs.annotations.NonNull;
26  
27  public class DefaultControlSelectionFilter implements IControlSelectionFilter {
28    private static final Logger LOGGER = LogManager.getLogger(DefaultControlSelectionFilter.class);
29  
30    @NonNull
31    private final List<Selection> selections;
32  
33    /**
34     * Construct a new selection filter based on the provided list of select
35     * criteria.
36     *
37     * @param selections
38     *          a list of select criteria
39     */
40    @SuppressWarnings("null")
41    public DefaultControlSelectionFilter(@NonNull List<? extends IProfileSelectControlById> selections) {
42      this.selections = selections.stream()
43          // ignore null entries
44          .filter(Objects::nonNull)
45          // create a selection object for the selection
46          .map(Selection::new)
47          .collect(Collectors.toUnmodifiableList());
48    }
49  
50    @NonNull
51    @Override
52    public Pair<Boolean, Boolean> apply(IControl control) {
53      String id = control.getId();
54      if (id == null) {
55        throw new ProfileResolutionEvaluationException("control is missing an identifier");
56      }
57      return match(id);
58    }
59  
60    /**
61     * Checks if the provided control identifier matches the criteria defined by
62     * this object.
63     *
64     * @param id
65     *          the control identifier to match
66     * @return a {@link Pair} whose first member is {@code true} for a match or
67     *         {@code false} otherwise, and whose second member is {@code true} if
68     *         the match applies to any child controls or {@code false} otherwise
69     */
70    @SuppressWarnings("null")
71    @NonNull
72    protected Pair<Boolean, Boolean> match(String id) {
73      return selections.parallelStream()
74          .map(selection -> selection.match(id))
75          // filter out non-matches
76          .filter(Pair::getLeft)
77          // aggregate matches
78          .reduce((first, second) -> {
79            Pair<Boolean, Boolean> result;
80            if (first.getLeft() || second.getLeft()) {
81              // at least one matches
82              boolean withChild = first.getLeft() && first.getRight() || second.getLeft() && second.getRight();
83              result = Pair.of(true, withChild);
84            } else {
85              result = NON_MATCH;
86            }
87            return result;
88          })
89          .orElse(NON_MATCH);
90    }
91  
92    @SuppressWarnings("PMD.ImplicitSwitchFallThrough")
93    private static Pattern toPattern(@NonNull Matching matching) {
94      String pattern = ObjectUtils.requireNonNull(matching.getPattern());
95      String regex = pattern.chars().boxed().map(ch -> (char) ch.intValue()).map(ch -> {
96  
97        String value;
98        switch (ch) {
99        case '*':
100         value = ".*";
101         break;
102       case '?':
103         value = ".";
104         break;
105       case '.':
106       case '+':
107       case '\\':
108       case '[':
109       case ']':
110       case '{':
111       case '}':
112       case '(':
113       case ')':
114       case '^':
115       case '$':
116         value = "\\" + ch;
117         break;
118       default:
119         value = String.valueOf(ch);
120       }
121       return value;
122     }).collect(Collectors.joining("", "^", "$"));
123 
124     if (LOGGER.isTraceEnabled()) {
125       LOGGER.atTrace().log("regex: {}", regex);
126     }
127     return Pattern.compile(regex);
128   }
129 
130   private static class Selection {
131 
132     private final boolean withChildControls;
133     private final Set<String> identifiers;
134     private final List<Pattern> patterns;
135 
136     public Selection(IProfileSelectControlById selection) {
137       // process with-child-controls
138       // default is "no"
139       this.withChildControls = "yes".equals(selection.getWithChildControls());
140 
141       // process with-ids
142       List<String> ids = selection.getWithIds();
143       if (ids == null) {
144         ids = Collections.emptyList();
145       }
146       this.identifiers = ids.stream()
147           .filter(Objects::nonNull)
148           .collect(Collectors.toUnmodifiableSet());
149 
150       // process with-ids
151       List<Matching> matching = selection.getMatching();
152       if (matching == null) {
153         matching = Collections.emptyList();
154       }
155       this.patterns = matching.stream()
156           .filter(Objects::nonNull)
157           .map(DefaultControlSelectionFilter::toPattern)
158           .collect(Collectors.toUnmodifiableList());
159     }
160 
161     public boolean isWithChildControls() {
162       return withChildControls;
163     }
164 
165     @NonNull
166     protected Pair<Boolean, Boolean> match(String id) {
167       // first check for direct match
168       boolean result = identifiers.stream().anyMatch(controlIdentifier -> controlIdentifier.equals(id));
169       if (!result) {
170         // next check for pattern match
171         result = patterns.stream().anyMatch(pattern -> pattern.asMatchPredicate().test(id));
172       }
173       return ObjectUtils.notNull(Pair.of(result, isWithChildControls()));
174     }
175   }
