26  import java.io.IOException;

28  import java.util.stream.Collectors;
29  import java.util.stream.Stream;
30  
31  class FilterNonSelectedVisitorTest {
32  
33    @SuppressWarnings("null")
34    @Test
35    void test() throws IOException {
36      // setup the imported catalog
37      IDocumentNodeItem importedCatalogDocumentItem = TestUtil.newImportedCatalog();
38  
39      // setup the selection visitor
40      IControlFilter filter = IControlFilter.newInstance(
41          IControlSelectionFilter.ALL_MATCH,
42          IControlSelectionFilter.matchIds("control2", "control5", "control7"));
43      IIndexer indexer = new BasicIndexer();
44  
45      IControlSelectionState state = new ControlSelectionState(indexer, filter);
46      // process selections
47      ControlSelectionVisitor.instance().visitCatalog(importedCatalogDocumentItem, state);
48  
49      // setup reference counting
50      ReferenceCountingVisitor.instance().visitCatalog(
51          importedCatalogDocumentItem,
52          indexer,
53          (uri, src) -> importedCatalogDocumentItem.getBaseUri().resolve(uri));
54  
55      FilterNonSelectedVisitor.instance().visitCatalog(importedCatalogDocumentItem, indexer);
56  
57      Set<String> selected = Stream.concat(
58          indexer.getEntitiesByItemType(IEntityItem.ItemType.GROUP).stream(),
59          indexer.getEntitiesByItemType(IEntityItem.ItemType.CONTROL).stream())
60          .filter(entry -> indexer.isSelected(entry))
61          .map(entry -> {
62            IControlContainer container = entry.getInstanceValue();
63            String id;
64            if (container instanceof Control) {
65              id = ((Control) container).getId();
66            } else if (container instanceof CatalogGroup) {
67              id = ((CatalogGroup) container).getId();
68            } else {
69              throw new UnsupportedOperationException(
70                  String.format("Invalid container type: %s", container.getClass().getName()));
71            }
72            return id;
73          })
74          .collect(Collectors.toSet());
75      assertEquals(
76          Set.of("control1", "control3", "control4", "control6", "control8", "group1", "group2"),
77          selected);
78  
79      OscalBindingContext.instance().newSerializer(Format.YAML, Catalog.class)
80          .serialize(
81              (Catalog) INodeItem.toValue(importedCatalogDocumentItem),
82              System.out);
83    }
84  }
