48    @NonNull
49    private IControlSelectionFilter newEmptyFilter() {

51        {
52          allowing(selectControlByIdA).getWithChildControls();
53          will(returnValue("no"));
54          allowing(selectControlByIdA).getWithIds();

62          ObjectUtils.notNull(List.of(selectControlByIdA)));

213   /**
214    * Test the filtering of multiple match criteria.
215    */
216   @Test
217   @SuppressFBWarnings("RV_RETURN_VALUE_IGNORED_NO_SIDE_EFFECT")
218   void testMultipleSelectionFilter() {
219     final IControl control1 = context.mock(IControl.class, "control1");
220     final IControl control2 = context.mock(IControl.class, "control2");
221     final IControl control3 = context.mock(IControl.class, "control3");
222     final IControl control4 = context.mock(IControl.class, "control4");
223     final IControl control5 = context.mock(IControl.class, "control5");
224     final IControl control6 = context.mock(IControl.class, "control6");
225     final IControl control7 = context.mock(IControl.class, "control7");
226     context.checking(new Expectations() {
227       {
228         allowing(control1).getId();
229         will(returnValue("test"));
230         allowing(control2).getId();
231         will(returnValue("test2"));
232         allowing(control3).getId();
233         will(returnValue("test4"));
234         allowing(control4).getId();
235         will(returnValue("other1"));
236         allowing(control5).getId();
237         will(returnValue("other"));
238         allowing(control6).getId();
239         will(returnValue("example1"));
240         allowing(control7).getId();
241         will(returnValue("example11"));
242       }
243     });
244 
245     IControlSelectionFilter filter = newMultipleSelectionFilter();
246     Pair<Boolean, Boolean> pair = filter.apply(control1);
247     assertFalse(pair.getLeft());
248     assertFalse(pair.getRight());
249 
250     pair = filter.apply(control2);
251     assertTrue(pair.getLeft());
252     assertTrue(pair.getRight());
253 
254     pair = filter.apply(control3);
255     assertTrue(pair.getLeft());
256     assertFalse(pair.getRight());
257 
258     // test match patterns
259     pair = filter.apply(control4);
260     assertTrue(pair.getLeft());
261     assertTrue(pair.getRight());
262 
263     pair = filter.apply(control5);
264     assertTrue(pair.getLeft());
265     assertTrue(pair.getRight());
266 
267     pair = filter.apply(control6);
268     assertTrue(pair.getLeft());
269     assertTrue(pair.getRight());
270 
271     pair = filter.apply(control7);
272     assertTrue(pair.getLeft());
273     assertFalse(pair.getRight());
274   }
