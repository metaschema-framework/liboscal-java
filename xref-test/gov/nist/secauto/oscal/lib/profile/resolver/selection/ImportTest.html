33  class ImportTest {
34  
35    @NonNull
36    private static IDocumentNodeItem newImportedCatalog() {
37  
38      // setup the imported catalog
39      Catalog importedCatalog = new Catalog();
40  
41      importedCatalog.addControl(AbstractControl.builder("control1")
42          .title("Control 1")
43          .build());
44      importedCatalog.addControl(AbstractControl.builder("control2")
45          .title("Control 2")
46          .build());
47  
48      return INodeItemFactory.instance().newDocumentNodeItem(
49          ObjectUtils.requireNonNull(
50              (IBoundDefinitionModelAssembly) OscalBindingContext.instance().getBoundDefinitionForClass(Catalog.class)),
51          ObjectUtils.notNull(Paths.get(System.getProperty("user.dir")).toUri()),
52          importedCatalog);
53    }
54  
55    @SuppressWarnings("null")
56    @Test
57    void test() throws ProfileResolutionException {
58      URI cwd = Paths.get(System.getProperty("user.dir")).toUri();
59  
60      ProfileImport profileImport = new ProfileImport();
61      profileImport.setIncludeAll(new IncludeAll());
62      profileImport.setExcludeControls(Collections.singletonList(
63          AbstractProfileSelectControlById.builder()
64              .withId("control1")
65              .build()));
66      profileImport.setHref(cwd);
67  
68      // setup the profile
69      Profile profile = new Profile();
70  
71      profile.addImport(profileImport);
72  
73      IDocumentNodeItem profileDocumentItem = INodeItemFactory.instance().newDocumentNodeItem(
74          ObjectUtils.requireNonNull(
75              (IBoundDefinitionModelAssembly) OscalBindingContext.instance().getBoundDefinitionForClass(Profile.class)),
76          cwd,
77          profile);
78      // setup the imported catalog
79      IDocumentNodeItem importedCatalogDocumentItem = newImportedCatalog();
80  
81      // setup the resolved catalog
82      Catalog resolvedCatalog = new Catalog();
83      for (IRootAssemblyNodeItem profileRootItem : CollectionUtil
84          .toIterable(profileDocumentItem.getModelItemsByName(OscalModelConstants.QNAME_PROFILE).stream()
85              .map(rootItem -> (IRootAssemblyNodeItem) rootItem))) {
86        for (IAssemblyNodeItem importItem : CollectionUtil.toIterable(
87            profileRootItem.getModelItemsByName(OscalModelConstants.QNAME_IMPORT).stream()
88                .map(item -> (IAssemblyNodeItem) item))) {
89  
90          Import catalogImport = new Import(profileRootItem, importItem);
91          catalogImport.resolve(
92              importedCatalogDocumentItem,
93              resolvedCatalog,
94              (uri, src) -> importedCatalogDocumentItem.getBaseUri().resolve(uri));
95        }
96  
97      }
98  
99    }
