71  
72    private static String metapath(@NonNull IDefinitionNodeItem<?, ?> item) {
73      return metapath(item.getMetapath());
74    }
75  
76    private static String metapath(@NonNull String path) {
77      return path.replace("[1]", "");
78    }
79  
80    private static String values(@NonNull IAllowedValuesConstraint constraint) {
81      return constraint.getAllowedValues().values().stream()
82          .map(IAllowedValue::getValue)
83          .collect(Collectors.joining(", "));
84    }
85  
86    @NonNull
87    private static <K, V> NavigableMap<K, V> sortMap(@NonNull Map<K, V> map, @NonNull Comparator<? super K> comparator) {
88      TreeMap<K, V> retval = new TreeMap<>(comparator);
89      retval.putAll(map);
90      return retval;
91    }
