29  class AbstractNodeItemVisitorTest {
30  
31    @Test
32    void testAllowedValues() {
33      IBoundModule module = OscalBindingContext.instance().registerModule(OscalCompleteModule.class);
34      AllowedValueCollectingNodeItemVisitor walker = new AllowedValueCollectingNodeItemVisitor();
35      walker.visit(module);
36  
37      System.out.println();
38      System.out.println("Allowed Values");
39      System.out.println("--------------");
40      Map<IDefinitionNodeItem<?, ?>,
41          List<AllowedValueCollectingNodeItemVisitor.AllowedValuesRecord>> allowedValuesByTarget
42              = walker.getAllowedValueLocations().stream()
43                  .flatMap(location -> location.getAllowedValues().stream())
44                  .collect(Collectors.groupingBy(AllowedValuesRecord::getTarget,
45                      LinkedHashMap::new,
46                      Collectors.mapping(Function.identity(), Collectors.toUnmodifiableList())));
47      sortMap(allowedValuesByTarget, Comparator.comparing(IDefinitionNodeItem::getMetapath)).entrySet().stream()
48          .forEachOrdered(entry -> {
49            IDefinitionNodeItem<?, ?> target = entry.getKey();
50            List<AllowedValueCollectingNodeItemVisitor.AllowedValuesRecord> allowedValuesRecords = entry.getValue();
51  
52            System.out.println("node:             " + metapath(target));
53            allowedValuesRecords.forEach(record -> {
54              assert target.equals(record.getTarget());
55  
56              IAllowedValuesConstraint constraint = record.getAllowedValues();
57              System.out.println("- allowed-values:");
58              System.out.println("    location:     " + metapath(record.getLocation()));
59              System.out.println("    target:       " + constraint.getTarget());
60              System.out.println("    values:       " + values(constraint));
61              System.out.println("    allow-other:  " + constraint.isAllowedOther());
62              // System.out.println(" extensible: " + constraint.getExtensible());
63              // System.out.println(" source: " + constraint.getSource());
64            });
65          });
66    }
67  
68    private static String metapath(@NonNull IDefinitionNodeItem<?, ?> item) {
69      return metapath(item.getMetapath());
70    }
71  
72    private static String metapath(@NonNull String path) {
73      return path.replaceAll("\\[1\\]", "");
74    }
75  
76    private static String values(@NonNull IAllowedValuesConstraint constraint) {
77      return constraint.getAllowedValues().values().stream()
78          .map(IAllowedValue::getValue)
79          .collect(Collectors.joining(", "));
80    }
81  
82    @NonNull
83    private static <K, V> NavigableMap<K, V> sortMap(@NonNull Map<K, V> map, @NonNull Comparator<? super K> comparator) {
84      TreeMap<K, V> retval = new TreeMap<>(comparator);
85      retval.putAll(map);
86      return retval;
87    }
