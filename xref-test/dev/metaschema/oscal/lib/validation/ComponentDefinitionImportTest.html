6   package dev.metaschema.oscal.lib.validation;
7   
8   import static org.junit.jupiter.api.Assertions.assertDoesNotThrow;
9   import static org.junit.jupiter.api.Assertions.assertInstanceOf;
10  import static org.junit.jupiter.api.Assertions.assertNotNull;
11  
12  import org.apache.logging.log4j.LogManager;
13  import org.apache.logging.log4j.Logger;
14  import org.junit.jupiter.api.Test;
15  
16  import java.io.IOException;
17  import java.nio.file.Path;
18  import java.nio.file.Paths;
19  
20  import dev.metaschema.core.model.validation.IValidationResult;
21  import dev.metaschema.oscal.lib.OscalBindingContext;
22  import dev.metaschema.oscal.lib.model.ComponentDefinition;
23  import dev.metaschema.oscal.lib.model.IOscalInstance;
24  
25  /**
26   * Test case for oscal-cli issue #192: Validates that ComponentDefinition
27   * implements IOscalInstance, preventing ClassCastException when validating
28   * component definitions with import-component-definitions.
29   *
30   * @see <a href=
31   *      "https://github.com/metaschema-framework/oscal-cli/issues/192">Issue
32   *      #192</a>
33   */
34  class ComponentDefinitionImportTest {
35    private static final Logger LOGGER = LogManager.getLogger(ComponentDefinitionImportTest.class);
36  
37    /**
38     * Verifies that ComponentDefinition implements IOscalInstance. This is required
39     * for the resolve-reference() function to work correctly in
40     * ResolveReference.resolveReference() at line 152.
41     */
42    @Test
43    void testComponentDefinitionImplementsIOscalInstance() throws IOException {
44      OscalBindingContext bindingContext = OscalBindingContext.newInstance();
45  
46      Path basePath = Paths.get("src/test/resources/content/issue-192/base-component-definition.json");
47      ComponentDefinition componentDef = bindingContext.loadComponentDefinition(basePath);
48  
49      assertInstanceOf(IOscalInstance.class, componentDef,
50          "ComponentDefinition must implement IOscalInstance for resolve-reference() to work");
51  
52      LOGGER.info("ComponentDefinition correctly implements IOscalInstance");
53    }
54  
55    /**
56     * Verifies that validating a ComponentDefinition with
57     * import-component-definitions does not throw a ClassCastException.
58     *
59     * Previously (issue #192), this would fail with: ClassCastException: class
60     * dev.metaschema.oscal.lib.model.ComponentDefinition cannot be cast to class
61     * dev.metaschema.oscal.lib.model.IOscalInstance
62     */
63    @Test
64    void testValidateComponentDefinitionWithImport() {
65      OscalBindingContext bindingContext = OscalBindingContext.newInstance();
66  
67      Path componentDefPath = Paths.get("src/test/resources/content/issue-192/component-definition-with-import.json");
68  
69      // Validation should complete without ClassCastException
70      IValidationResult validationResult = assertDoesNotThrow(
71          () -> bindingContext.validateWithConstraints(componentDefPath.toUri(), null),
72          "Validation should not throw ClassCastException");
73  
74      assertNotNull(validationResult, "Validation result should not be null");
75  
76      if (validationResult.isPassing()) {
77        LOGGER.info("Validation passed successfully");
78      } else {
79        LOGGER.info("Validation completed with {} findings (no ClassCastException)",
80            validationResult.getFindings().size());
81      }
82    }
83  }
